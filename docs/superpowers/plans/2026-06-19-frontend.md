# Трекер недвижимости — Plan 3: Frontend

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Next.js веб-приложение с 7 экранами: лендинг, авторизация, дашборд, создание фильтра, список объявлений, карточка объявления, настройки.

**Architecture:** Next.js 14 App Router. Лендинг — серверный рендеринг. Всё за авторизацией — клиентские компоненты. JWT хранится в httpOnly cookie. API-клиент — обёртка над fetch.

**Tech Stack:** Next.js 14, TypeScript, Tailwind CSS, Recharts (графики), react-select (мультиселект), js-cookie

## Global Constraints

- Next.js >= 14, Node >= 20
- TypeScript strict mode
- Tailwind — без UI-библиотек (MUI, shadcn и т.д.)
- JWT хранить в httpOnly cookie, не в localStorage
- История цен на карточке — только если пользователь basic/pro, иначе блур + CTA
- Ссылка на оригинал объявления открывается в новой вкладке
- OAuth redirect — стандартный server-side flow через `/auth/oauth/{provider}`

import Cookies from "js-cookie";

const TOKEN_KEY = "rt_token";

export function getToken(): string | null {
  return Cookies.get(TOKEN_KEY) ?? null;
}

export function setToken(token: string): void {
  Cookies.set(TOKEN_KEY, token, { expires: 7, sameSite: "lax" });
}

export function clearToken(): void {
  Cookies.remove(TOKEN_KEY);
}

export function isAuthenticated(): boolean {
  return !!getToken();
}

- [ ] **Шаг 4: `frontend/src/lib/api.ts`**

```typescript
import { getToken } from "./auth";
import type {
  TokenResponse, Filter, FilterCreate,
  Listing, ListingDetail, PricePoint,
  Subscription, Account,
} from "@/types";

const BASE = process.env.NEXT_PUBLIC_API_URL ?? "http://localhost:8000";

async function request<T>(
  path: string,
  options: RequestInit = {}
): Promise<T> {
  const token = getToken();
  const res = await fetch(`${BASE}${path}`, {
    ...options,
    headers: {
      "Content-Type": "application/json",
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
      ...options.headers,
    },
  });
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err.detail ?? `HTTP ${res.status}`);
  }
  if (res.status === 204) return undefined as T;
  return res.json();
}

export const api = {
  auth: {
    register: (email: string, password: string) =>
      request<TokenResponse>("/auth/register", {
        method: "POST",
        body: JSON.stringify({ email, password }),
      }),
    login: (email: string, password: string) =>
      request<TokenResponse>("/auth/login", {
        method: "POST",
        body: JSON.stringify({ email, password }),
      }),
    oauthUrl: (provider: string) => `${BASE}/auth/oauth/${provider}`,
  },

  filters: {
    list: () => request<Filter[]>("/filters"),
    create: (data: FilterCreate) =>
      request<Filter>("/filters", { method: "POST", body: JSON.stringify(data) }),
    update: (id: string, data: FilterCreate) =>
      request<Filter>(`/filters/${id}`, { method: "PUT", body: JSON.stringify(data) }),
    delete: (id: string) =>
      request<void>(`/filters/${id}`, { method: "DELETE" }),
    toggle: (id: string) =>
      request<Filter>(`/filters/${id}/toggle`, { method: "PATCH" }),
  },

  listings: {
    byFilter: (filterId: string, sort = "date", type = "all") =>
      request<Listing[]>(`/filters/${filterId}/listings?sort=${sort}&type=${type}`),
    get: (id: string) => request<ListingDetail>(`/listings/${id}`),
    history: (id: string) => request<PricePoint[]>(`/listings/${id}/history`),
  },

  subscription: {
    get: () => request<Subscription>("/subscription"),
    checkout: (plan: string) =>
      request<{ confirmation_url: string }>("/subscription/checkout", {
        method: "POST",
        body: JSON.stringify({ plan }),
      }),
  },

  account: {
    get: () => request<Account>("/account"),
    updateAlertEmail: (email: string) =>
      request<Account>("/account/alert-email", {
        method: "PUT",
        body: JSON.stringify({ email }),
      }),
  },
};
```

import { useEffect } from "react";
import { useRouter } from "next/navigation";
import { isAuthenticated } from "@/lib/auth";

export default function AuthGuard({ children }: { children: React.ReactNode }) {
  const router = useRouter();
  useEffect(() => {
    if (!isAuthenticated()) router.replace("/login");
  }, [router]);
  if (!isAuthenticated()) return null;
  return <>{children}</>;
}

    window.location.href = confirmation_url;
  }

  if (!account || !sub) return <AuthGuard><div className="p-10 text-center text-gray-400">Загрузка...</div></AuthGuard>;

  return (
    <AuthGuard>
      <main className="max-w-xl mx-auto px-4 py-10 space-y-8">
        <h1 className="text-xl font-bold">Настройки</h1>

        {/* Аккаунт */}
        <section className="border border-gray-200 rounded-xl p-6">
          <h2 className="font-semibold mb-4">Аккаунт</h2>
          <p className="text-sm text-gray-500 mb-4">Email: {account.email}</p>
          <form onSubmit={handleSaveEmail} className="flex gap-3">
            <input
              type="email"
              value={alertEmail}
              onChange={(e) => setAlertEmail(e.target.value)}
              placeholder="Email для алертов"
              className="flex-1 border border-gray-300 rounded-lg px-4 py-2 text-sm outline-none focus:border-black"
            />
            <button
              type="submit"
              className="bg-black text-white px-4 py-2 rounded-lg text-sm"
            >
              {saved ? "Сохранено ✓" : "Сохранить"}
            </button>
          </form>
        </section>

        {/* Подписка */}
        <section className="border border-gray-200 rounded-xl p-6">
          <h2 className="font-semibold mb-2">Тариф</h2>
          <p className="text-sm text-gray-500 mb-4">
            Текущий: <strong>{PLAN_LABELS[sub.plan]}</strong>
            {sub.expires_at && ` · до ${new Date(sub.expires_at).toLocaleDateString("ru-RU")}`}
          </p>

          {sub.plan === "free" && (
            <div className="grid grid-cols-2 gap-3">
              <div className="border rounded-xl p-4">
                <div className="font-medium mb-1">Базовый</div>
                <div className="text-lg font-bold mb-3">299 ₽/мес</div>
                <ul className="text-xs text-gray-500 space-y-1 mb-4">
                  <li>— 5 фильтров</li>
                  <li>— Мгновенные алерты</li>
                  <li>— История 30 дней</li>
                </ul>
                <button
                  onClick={() => handleUpgrade("basic")}
                  className="w-full bg-black text-white rounded-lg py-2 text-sm"
                >
                  Выбрать
                </button>
              </div>
              <div className="border border-black rounded-xl p-4">
                <div className="font-medium mb-1">Про</div>
                <div className="text-lg font-bold mb-3">599 ₽/мес</div>
                <ul className="text-xs text-gray-500 space-y-1 mb-4">
                  <li>— 20 фильтров</li>
                  <li>— Мгновенные алерты</li>
                  <li>— Полная история</li>
                </ul>
                <button
                  onClick={() => handleUpgrade("pro")}
                  className="w-full bg-black text-white rounded-lg py-2 text-sm"
                >
                  Выбрать
                </button>
              </div>
            </div>
          )}

          {sub.plan !== "free" && (
            <p className="text-sm text-gray-400">
              Для отмены подписки напишите на support@example.com
            </p>
          )}
        </section>
      </main>
    </AuthGuard>
  );
}
