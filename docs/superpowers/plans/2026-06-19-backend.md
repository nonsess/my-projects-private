# Трекер недвижимости — Plan 2: Backend API

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** FastAPI-бэкенд с авторизацией (email+OAuth), CRUD фильтров, API листингов с историей цен, матчингом фильтров, email-алертами и подпиской ЮKassa.

**Architecture:** FastAPI + SQLAlchemy 2.0 (async) + Alembic. Celery-воркер (shared с парсером Redis) обрабатывает матчинг и отправку алертов. Авторизация через JWT + OAuth (Google/VK/Яндекс) через Authlib.

**Tech Stack:** Python 3.12, FastAPI, SQLAlchemy 2.0 async, Alembic, Authlib, python-jose, Postmark/Resend, ЮKassa SDK, Celery, Redis, PostgreSQL 16

## Global Constraints

- Python >= 3.12
- Все эндпоинты защищены JWT (кроме /auth/*, /webhooks/*)
- Тарифные лимиты: free=1 фильтр/задержка 24ч, basic=5 фильтров/мгновенно, pro=20/мгновенно+полная история
- OAuth провайдеры: google, vk, yandex
- Платежи только ЮKassa
- Парсер и бэкенд используют одну PostgreSQL БД — бэкенд добавляет свои таблицы через Alembic
- Алерты: дедупликация — одно объявление не отправляется дважды по одному фильтру

- [ ] **Шаг 3: `backend/app/services/auth.py`**

```python
from datetime import datetime, timedelta, timezone
from jose import jwt, JWTError
from passlib.context import CryptContext
from app.config import settings

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def hash_password(password: str) -> str:
    return pwd_context.hash(password)


def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)


def create_access_token(user_id: str) -> str:
    expire = datetime.now(timezone.utc) + timedelta(
        minutes=settings.access_token_expire_minutes
    )
    return jwt.encode(
        {"sub": user_id, "exp": expire},
        settings.secret_key,
        algorithm="HS256",
    )


def decode_token(token: str) -> str:
    payload = jwt.decode(token, settings.secret_key, algorithms=["HS256"])
    user_id = payload.get("sub")
    if not user_id:
        raise JWTError("no sub")
    return user_id
```

```python
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from jose import JWTError
from app.database import get_session
from app.models.user import User
from app.services.auth import decode_token

bearer = HTTPBearer()


async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(bearer),
    session: AsyncSession = Depends(get_session),
) -> User:
    try:
        user_id = decode_token(credentials.credentials)
    except JWTError:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid token")

    result = await session.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()
    if not user:
        raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="User not found")
    return user
```

- [ ] **Шаг 6: `backend/app/routers/auth.py`**

```python
from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.database import get_session
from app.models.user import User
from app.models.subscription import Subscription
from app.schemas.auth import RegisterRequest, LoginRequest, TokenResponse
from app.services.auth import hash_password, verify_password, create_access_token

router = APIRouter()


@router.post("/register", response_model=TokenResponse)
async def register(data: RegisterRequest, session: AsyncSession = Depends(get_session)):
    result = await session.execute(select(User).where(User.email == data.email))
    if result.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="Email already registered")

    user = User(email=data.email, password_hash=hash_password(data.password))
    session.add(user)
    await session.flush()

    # Создать free подписку сразу
    session.add(Subscription(user_id=user.id, plan="free"))
    await session.commit()

    return TokenResponse(access_token=create_access_token(str(user.id)))


@router.post("/login", response_model=TokenResponse)
async def login(data: LoginRequest, session: AsyncSession = Depends(get_session)):
    result = await session.execute(select(User).where(User.email == data.email))
    user = result.scalar_one_or_none()
    if not user or not user.password_hash or not verify_password(data.password, user.password_hash):
        raise HTTPException(status_code=401, detail="Invalid credentials")

    return TokenResponse(access_token=create_access_token(str(user.id)))
```

```python
        _send_and_record(session, f, listing, event, alert_email)
        session.commit()


def _send_and_record(session, f, listing, event, alert_email):
    from app.models.alert import Alert
    from app.services.alerts import send_alert_email

    alert = Alert(
        user_id=f.user_id,
        filter_id=f.id,
        listing_id=listing.id,
        type=event,
    )
    session.add(alert)

    send_alert_email(
        to_email=alert_email,
        listing_url=listing.url,
        price=listing.price,
        address=listing.address,
        event=event,
    )
```

- [ ] **Шаг 2: `backend/app/routers/subscription.py`**

```python
import json
from datetime import datetime, timedelta, timezone
from fastapi import APIRouter, Depends, HTTPException, Request
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.database import get_session
from app.dependencies import get_current_user
from app.models.user import User
from app.models.subscription import Subscription
from app.services.subscription import create_checkout
from app.config import settings

router = APIRouter()


@router.get("/subscription")
async def get_subscription(
    user: User = Depends(get_current_user),
    session: AsyncSession = Depends(get_session),
):
    result = await session.execute(select(Subscription).where(Subscription.user_id == user.id))
    sub = result.scalar_one_or_none()
    if not sub:
        return {"plan": "free", "status": "active", "expires_at": None}
    return {"plan": sub.plan, "status": sub.status, "expires_at": sub.expires_at}


@router.post("/subscription/checkout")
async def checkout(
    body: dict,
    user: User = Depends(get_current_user),
):
    plan = body.get("plan")
    if plan not in ("basic", "pro"):
        raise HTTPException(status_code=400, detail="Invalid plan")
    return_url = f"{settings.frontend_url}/settings?payment=success"
    url = create_checkout(plan, str(user.id), return_url)
    return {"confirmation_url": url}


@router.post("/webhooks/yukassa")
async def yukassa_webhook(request: Request, session: AsyncSession = Depends(get_session)):
    body = await request.json()
    if body.get("event") != "payment.succeeded":
        return {"ok": True}

    metadata = body.get("object", {}).get("metadata", {})
    user_id = metadata.get("user_id")
    plan = metadata.get("plan")

    if not user_id or not plan:
        return {"ok": True}

    result = await session.execute(select(Subscription).where(Subscription.user_id == user_id))
    sub = result.scalar_one_or_none()

    expires = datetime.now(timezone.utc) + timedelta(days=30)
    if sub:
        sub.plan = plan
        sub.status = "active"
        sub.expires_at = expires
    else:
        session.add(Subscription(user_id=user_id, plan=plan, status="active", expires_at=expires))

    await session.commit()
    return {"ok": True}
```
